Send Single SMS — POST /sms/send
curl -X POST {YOUR_URL}/sms/send \ -H "x-api-key: sms_your_key" \ -H "Content-Type: application/json" \ -d '{"to":"+1234567890","message":"Hello!"}'
JavaScript
const res = await fetch('{YOUR_URL}/sms/send', { method: 'POST', headers: { 'x-api-key': 'sms_key', 'Content-Type': 'application/json' }, body: JSON.stringify({ to: '+1234567890', message: 'Hello!' }) }); const data = await res.json();
Python
import requests r = requests.post('{YOUR_URL}/sms/send', headers={'x-api-key': 'sms_key'}, json={'to': '+1234567890', 'message': 'Hello!'} ) print(r.json())
